 def RidgeGradientDescent(x, y, alpha, iters, L):
    x=np.matrix(x)
    #y=np.matrix(y)
    y=np.matrix(y).transpose()
    m, n = np.shape(x)
    beta = np.matrix(np.ones(n)).transpose()
    XT = x.transpose()
    MSES=[]
    for i in range(0, iters):
        y_hat = np.dot(x, beta)
        residuals = y-y_hat
        MSE = (residuals.transpose()*residuals)/len(x)
        MSES.append(MSE)
        #print("iteration:", i, "MSE:", MSE)
        ols_gradient = np.dot(XT, residuals) / m
        beta = beta - alpha * (ols_gradient + (L/m)*beta)
    return beta,MSES

def RidgeAutoTune(x,y,alpha):
    #i=L1
    lam=[.01,.1,1,10,100,1000,10000]
    x=np.matrix(x)
    #y=np.matrix(y)
    y=np.matrix(y).transpose()
    XTX = x.transpose()*x
    XTY = x.transpose()*y
    MSES = []
    betas = []
    LS = []
    #while i < L2:
    for la in lam:
        L = la
        LI = L*np.matrix(np.identity(len(XTX)))
        beta = linalg.inv(XTX-LI)*XTY
        y_hat = np.dot(x,beta)
        residuals = y-y_hat
        SSE = residuals.transpose()*residuals
        MSE = SSE/len(x)
        MSES.append(MSE)
        betas.append(beta)
        LS.append(L)
    MSES = np.squeeze(MSES)
    LMSE = np.vstack((LS,MSES)).T
    BV = pd.DataFrame(LMSE)
    BV.columns=['Lambda','MSE']
    srt = BV.sort_values(by=['MSE'], ascending=True, inplace=True)
    optimalL = BV['Lambda']
    print("the optimal Lambda is")
    return optimalL, LMSE, BV
